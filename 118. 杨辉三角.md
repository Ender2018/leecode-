class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if numRows == 0:
            return []
        res = [[1]]
        for i in range(1, numRows):
            tmp = [1, 1]
            for j in range(len(res[-1])-1):
                tmp.insert(1, res[-1][j] + res[-1][j+1])
            res.append(tmp)
        return res
