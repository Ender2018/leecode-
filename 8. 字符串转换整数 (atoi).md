class Solution:
    def myAtoi(self, str: str) -> int:
        str = str.lstrip()
        if str == '':
            return 0
        ans, flag = 0, 1
        if str[0] == '-':
            flag = -1
        elif str[0] == '+':
            flag = 1
        elif str[0].isdigit() == False:
            return 0
        else:
            ans = int(str[0])
        for c in str[1:]:
            if c.isdigit() == False:
                break
            if c >= '0' and c <= '9':
                if flag == 1 and (ans > (2**31-1)//10 or (ans == (2**31-1)//10 and int(c) > 7)):
                    return 2**31-1
                elif flag == -1 and (ans > 2**31//10 or (ans == 2**31//10 and int(c) > 8)):
                    return -2**31
                else:
                    ans = ans*10 + int(c)
        return ans*flag
